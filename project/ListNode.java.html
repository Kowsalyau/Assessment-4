public class ListNode {
    // Information part (node's data)
    private int data;

    // Reference to the next node
    private ListNode next;

    // Constructor to initialize the node with data (without next node reference)
    public ListNode(int data) {
        this.data = data;
        this.next = null;  // By default, next is null when creating a new node
    }

    // Constructor to initialize both data and the next node reference
    public ListNode(int data, ListNode next) {
        this.data = data;
        this.next = next;
    }

    // Getter for the data
    public int getData() {
        return data;
    }

    // Setter for the data (with validation to prevent negative values)
    public void setData(int data) {
        if (data >= 0) {  // Only allow non-negative values
            this.data = data;
        }
    }

    // Getter for the next node reference
    public ListNode getNext() {
        return next;
    }

    // Setter for the next node reference
    public void setNext(ListNode next) {
        this.next = next;
    }

    // Method to check if this node is the last one (next is null)
    public boolean isLast() {
        return next == null;
    }

    // Override toString() to display the data of the node
    @Override
    public String toString() {
        return "ListNode{data=" + data + "}";
    }

    // Main method for testing
    public static void main(String[] args) {
        // Create nodes
        ListNode node1 = new ListNode(10);
        ListNode node2 = new ListNode(20, node1);
        ListNode node3 = new ListNode(30, node2);

        // Print each node's data
        System.out.println(node3);  // ListNode{data=30}
        System.out.println(node2);  // ListNode{data=20}
        System.out.println(node1);  // ListNode{data=10}

        // Check if node1 is the last node
        System.out.println("Is node1 the last node? " + node1.isLast());  // true
        System.out.println("Is node2 the last node? " + node2.isLast());  // false
    }
}
